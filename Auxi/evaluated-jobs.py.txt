#!/usr/bin/env python3
"""Helpers for evaluated-jobs.csv deduplication."""

from __future__ import annotations

import argparse
import csv
import re
import sys
from pathlib import Path

CSV_PATH = Path(__file__).resolve().parent.parent / "data" / "evaluated-jobs.csv"
FIELDS = ["Company", "Role", "Base", "ExpPenalty", "PlatformPenalty", "CompPenalty", "AIBonus", "Score", "Applied", "Platform", "URL", "Source"]


def normalize_key(company: str, role: str) -> str:
    def norm(value: str) -> str:
        value = value.lower().strip()
        value = re.sub(r"^the\s+", "", value)
        value = re.sub(r"[^a-z0-9&]+", " ", value)
        return re.sub(r"\s+", " ", value).strip()

    return f"{norm(company)}|{norm(role)}"


def extract_job_id(url: str) -> str | None:
    if not url:
        return None
    # Greenhouse: ?gh_jid=1234567 or /jobs/1234567
    m = re.search(r"[?&]gh_jid=(\d+)", url)
    if m:
        return m.group(1)
    m = re.search(r"/jobs/(\d+)", url)
    if m:
        return m.group(1)
    # Lever / Ashby: UUID in path
    m = re.search(r"/([a-f0-9]{8}-[a-f0-9]{4}-[a-f0-9]{4}-[a-f0-9]{4}-[a-f0-9]{12})", url)
    if m:
        return m.group(1)
    return None


def read_rows() -> list[dict[str, str]]:
    with CSV_PATH.open(newline="", encoding="utf-8") as handle:
        return list(csv.DictReader(handle))


def write_rows(rows: list[dict[str, str]]) -> None:
    import shutil

    # Safety check: refuse to write if new row count is suspiciously low compared to existing file.
    if CSV_PATH.exists():
        existing_lines = sum(1 for _ in CSV_PATH.open(encoding="utf-8")) - 1  # subtract header
        if existing_lines > 10 and len(rows) < existing_lines // 2:
            raise RuntimeError(
                f"Refusing to write: new row count ({len(rows)}) is less than half of "
                f"existing row count ({existing_lines}). Something is wrong."
            )

    tmp = CSV_PATH.with_suffix(".tmp")
    with tmp.open("w", newline="", encoding="utf-8") as handle:
        writer = csv.DictWriter(handle, fieldnames=FIELDS, extrasaction="ignore")
        writer.writeheader()
        writer.writerows(rows)
    shutil.move(str(tmp), str(CSV_PATH))


def dedupe_rows(rows: list[dict[str, str]]) -> tuple[list[dict[str, str]], list[str]]:
    seen_keys: dict[str, int] = {}
    seen_ids: dict[str, int] = {}
    deduped: list[dict[str, str]] = []
    removed: list[str] = []

    for row in rows:
        key = normalize_key(row["Company"], row["Role"])
        job_id = extract_job_id(row.get("URL", ""))
        label = f'{row["Company"]} — {row["Role"]}'

        match_idx = seen_keys.get(key) if key in seen_keys else seen_ids.get(job_id) if job_id else None

        if match_idx is None:
            idx = len(deduped)
            seen_keys[key] = idx
            if job_id:
                seen_ids[job_id] = idx
            deduped.append(row)
            continue

        existing = deduped[match_idx]
        if (row.get("Applied") or "").strip().lower() == "yes" and (existing.get("Applied") or "").strip().lower() != "yes":
            removed.append(f"{label} (kept row with Applied=Yes)")
            deduped[match_idx] = row
        else:
            removed.append(label)

    return deduped, removed


def cmd_skip_list() -> int:
    rows = read_rows()
    for row in rows:
        print(f'{row["Company"]}\t{row["Role"]}')
    return 0


def cmd_is_new(company: str, role: str, url: str = "") -> int:
    key = normalize_key(company, role)
    incoming_id = extract_job_id(url)
    for row in read_rows():
        if normalize_key(row["Company"], row["Role"]) == key:
            print(f'ALREADY LOGGED: {row["Company"]} — {row["Role"]}', file=sys.stderr)
            return 1
        if incoming_id and extract_job_id(row.get("URL", "")) == incoming_id:
            print(f'ALREADY LOGGED (URL match): {row["Company"]} — {row["Role"]}', file=sys.stderr)
            return 1
    print(f'NEW: {company} — {role}')
    return 0


def cmd_is_new_batch() -> int:
    """Read tab-separated lines from stdin: Company<TAB>Role<TAB>URL (URL optional).

    Prints NEW or DUP per line. Duplicates within the batch itself are also caught.
    """
    rows = read_rows()
    keys = {normalize_key(r["Company"], r["Role"]) for r in rows}
    ids = set()
    for r in rows:
        job_id = extract_job_id(r.get("URL", ""))
        if job_id:
            ids.add(job_id)

    for line in sys.stdin:
        line = line.strip()
        if not line:
            continue
        parts = line.split("\t")
        if len(parts) < 2:
            print(f"SKIP (expected Company<TAB>Role<TAB>URL): {line}")
            continue
        company, role = parts[0].strip(), parts[1].strip()
        url = parts[2].strip() if len(parts) > 2 else ""
        key = normalize_key(company, role)
        job_id = extract_job_id(url)
        if key in keys or (job_id and job_id in ids):
            print(f"DUP: {company} - {role}")
        else:
            print(f"NEW: {company} - {role}")
            keys.add(key)
            if job_id:
                ids.add(job_id)
    return 0


def cmd_dedupe() -> int:
    rows = read_rows()
    deduped, removed = dedupe_rows(rows)

    if not removed:
        print("No duplicates found.")
        return 0

    write_rows(deduped)
    print(f"Removed {len(removed)} duplicate(s):")
    for item in removed:
        print(f"  - {item}")
    return 0


def main() -> int:
    parser = argparse.ArgumentParser(description=__doc__)
    subparsers = parser.add_subparsers(dest="command", required=True)

    subparsers.add_parser("skip-list", help="Print company/role pairs already in the CSV")

    is_new = subparsers.add_parser("is-new", help="Exit 0 if role is new, 1 if already logged")
    is_new.add_argument("company")
    is_new.add_argument("role")
    is_new.add_argument("url", nargs="?", default="", help="Optional job URL for ID-based dedup")

    subparsers.add_parser("is-new-batch",
                          help="Read Company<TAB>Role<TAB>URL lines from stdin, print NEW/DUP per line")

    subparsers.add_parser("dedupe", help="Remove duplicate rows from evaluated-jobs.csv")

    args = parser.parse_args()

    if args.command == "skip-list":
        return cmd_skip_list()
    if args.command == "is-new":
        return cmd_is_new(args.company, args.role, args.url)
    if args.command == "is-new-batch":
        return cmd_is_new_batch()
    if args.command == "dedupe":
        return cmd_dedupe()

    parser.print_help()
    return 1


if __name__ == "__main__":
    raise SystemExit(main())
